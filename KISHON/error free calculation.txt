import math


# 1. Basic Arithmetic Operations
def add(a,b):
    return a+b

def subtract(a,b):
    return a-b

def multiply(a,b):
    return a*b

def divide(a,b):
    if b!= 0:
        return a/b
    else:
        return "Error: Division by zero"

# 2. Find Logarithm
def find_log(value, base=10):
    return math.log(value, base)

# 3. Find Cosine
def find_cos(angle):
    return math.cos(math.radians(angle))

# 4. Time Conversion
def hours_to_minutes(hours):
    return hours*60

def minutes_to_hours(minutes):
    return minutes/60
