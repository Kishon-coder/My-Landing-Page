
def km_to_meter(km):
    return km*1000

def hours_to_minutes(hours):
    return hours*60

def kilo_to_gram(kg):
    return kg*1000

# Test the functions directly
print("5 kilometers =",km_to_meter(5),"meters")
print("2 hours =",hours_to_minutes(2),"minutes")
print("3 kilograms =",kilo_to_gram(3),"grams")




output

5 kilometers = 5000 meters
2 hours = 120 minutes
3 kilograms = 3000 grams
